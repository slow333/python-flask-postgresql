def enumerate(iterable, start=0):
    n = start
    for elem in iterable:
        yield n, elem
        n += 1

### break 문 : 자신이 직접 속한 for를 빠져나옴

prime number가 아닌 것에 대해...
for n in range(2, 10):
    for x in range(2, n):
        if n % x == 0:
            print(f"{n} equals {x} * {n//x}")
            break

continue 문 : 자신이 속한 for의 처음으로 돌아감, 뒤에 것을 실행안함

for num in range(2, 10):
    if num % 2 == 0:
        print(f"Found an even number {num}")
        continue
    print(f"Found an odd number {num}")

for, while에서의 else 문

for, while에서 break를 만나지 않으면 실행됨

for n in range(2, 10):
    for x in range(2, n):
        if n % x == 0:
            print(n, 'equals', x, '*', n//x)
            break
    else:
        # loop fell through without finding a factor
        print(n, 'is a prime number')
else:
    # loop fell through without finding a factor
    print('No prime numbers found')
