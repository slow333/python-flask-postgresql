readlines() 함수 사용하기

한글 깨지 방지를 위해 쓰기에 encoding=‘utf-8’ 필요
읽을 때도 encoding=‘utf-8’ 필요

f = open("새파일.txt", 'w', encoding='utf-8')
for i in range(1, 6):
    data = "%d번째 줄입니다.\n" % i
    f.write(data)
f.close()

f = open("새파일.txt", 'r', encoding='utf-8')
lines = f.readlines()
print(lines[len(lines) - 1])  # Print the last line
for line in lines:
    print(line)
f.close()

with문과 함께 사용하기(close()를 안해도 됨)

with open('external.txt', 'w') as f:
  f.write('Hello, World!')
with open('external.txt', 'r') as f:
  content = f.read()
  print(content)

pickle 변수를 파일로 저장하기

import pickle
f = open('external.txt', 'wb')
data = {1: 'python', 2: 'you need', '취미': ['축구', '골프', '코딩']}
pickle.dump(data, f)
f.close()

f = open('external.txt', 'rb')
data = pickle.load(f)
print(data)
f.close()
