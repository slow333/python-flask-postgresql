함수 사용 방법

def add_many(*args):
  result = 0
  for arg in args:
    result += arg
  return result

add_many(1,2,3,4,5,6,7)

def add_mul(choice, *args):
  if choice == "add":
    result = 0
    for arg in args:
      result += arg
  elif choice == "mul":
    result = 1
    for arg in args:
      result *= arg
  return result

add_mul('add', 1,2,3,4,5)
add_mul('mul', 1,2,3,4,5)

*args

나머지 매게변수로 동작함

**kwargs

입력값을 dicoanry 형태로 처리

return a, b 하면 2개의 값을 tuple로 반환함

def print_kwargs(**kwargs):
  print(kwargs)

print_kwargs(a=1, b=2, c=3, d=4) 

def add_and_mul(a,b):
  return a+b, a*b

add_and_mul(3,4)
result1, result2 = add_and_mul(3,4)
print(result1, result2)

전역변수 사용하기(local에서 global 사용)

a = 1
def vartest(a):
  global a
  a = a + 1

vartest(a)
print(a)
